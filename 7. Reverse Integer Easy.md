	class Solution {
	    public int reverse(int n) {
			int m = 0;
			int pos = 1;
			int a = Math.abs(n);
			while(a > 0) {
				m *= 10;
				m += a % 10;
				a = a /10;
				pos++;
				if(pos == 10) {
					if(n>0 && ((m > 214748364 && a !=0) || m == 214748364 && a>7) ){
						m = 0;
						break;
				}
					if(n<0 && ((m > 214748364 && a !=0) || m == 214748364 && a>8 )){
						m = 0;
						break;
					}
				}
			}
				
			if(m == 0) {
				return 0;
			}
			else {
				return m*(Math.abs(n)/n);
	        }
	    }
	}
